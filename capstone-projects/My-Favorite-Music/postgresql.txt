CREATE TABLE genres (
    id SERIAL PRIMARY KEY,
    deezer_id INT UNIQUE,
    name VARCHAR(50) NOT NULL,
    picture TEXT,
    picture_small TEXT,
    picture_medium TEXT,
    picture_big TEXT,
    picture_xl TEXT,
    type VARCHAR(50) NOT NULL
);

CREATE TABLE artists (
    id SERIAL PRIMARY KEY,
    deezer_id INT UNIQUE,
    name VARCHAR(50) NOT NULL,
    link VARCHAR(255),
    share VARCHAR(255),
    picture TEXT,
    picture_small TEXT,
    picture_medium TEXT,
    picture_big TEXT,
    picture_xl TEXT,
    nb_album INT,
    nb_fan INT,
    radio BOOLEAN NOT NULL,
    tracklist VARCHAR(300),
    type VARCHAR(50) NOT NULL,
    genre_id INT
);

CREATE TABLE albums (
    id SERIAL PRIMARY KEY,
    deezer_id INT UNIQUE,
    title VARCHAR(255) NOT NULL,
    upc VARCHAR(255),
    link VARCHAR(255),
    share VARCHAR(255),
    cover TEXT,
    cover_small TEXT,
    cover_medium TEXT,
    cover_big TEXT,
    cover_xl TEXT,
    md5_image TEXT,
    genre_id INT,
    genres JSON,
    label VARCHAR(255), 
    nb_tracks INT,
    duration INT,
    fans INT,
    release_date VARCHAR(50),
    record_type VARCHAR(50) NOT NULL,
    available BOOLEAN NOT NULL,
    tracklist VARCHAR(300),
    explicit_lyrics BOOLEAN NOT NULL,
    explicit_content_lyrics INT,
    explicit_content_cover INT,
    contributors JSON,
    artist JSON,
    type VARCHAR(100),
    tracks JSON
);

CREATE TABLE tracks (
    id SERIAL PRIMARY KEY,
    deezer_id BIGINT UNIQUE,
    readable BOOLEAN NOT NULL,
    title VARCHAR(255),
    title_short VARCHAR(255),
    title_version VARCHAR(255),
    isrc VARCHAR(255),
    link TEXT,
    share VARCHAR(255),
    duration INT,
    track_position INT,
    disk_number INT,
    rank INT,
    release_date VARCHAR(255),
    explicit_lyrics BOOLEAN NOT NULL,
    explicit_content_lyrics INT,
    explicit_content_cover INT,
    preview VARCHAR(255),
    bpm NUMERIC,
    gain NUMERIC,
    available_countries JSON,
    contributors JSON,
    md5_image VARCHAR(255),
    artist JSON,
    album JSON,
    type VARCHAR(255)
);
